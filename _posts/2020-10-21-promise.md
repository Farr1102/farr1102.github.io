### Promise 有三种状态

- pending
- fulfilld
- rejected

状态一旦改变不可逆

```js
// 定义promise状态
const PROMISE_PENDING_STATE = 'pending';
const PROMISE_FULFILLD_STATE = 'fulfilld';
const PROMISE_REJECTED_STATE = 'rejected';

Class MyPromise {
  constructor(execute) {
    this.PromiseState = PROMISE_PENDING_STATE;
    this.PromiseResult = undefined;
    this.callbacks = [];
    
    try {
      execute(this.resolve.bind(this), this.reject.bind(this));
    } catch(error) {
      this.reject(error);
    }
    
    resolve(result) {
      if (this.PromiseState === PROMISE_PENDING_STATE) {
        this.PromiseState = PROMISE_FULFILLED_STATE;
        this.PromiseResult = result;
        
        setTimeout(() => {
          this.callbacks.forEach(cb => {
            cb.onResolved(this.PromiseResult);
          })
        })
      }
    }
    reject(reason) {
      if (this.PromiseState === PROMISE_PENDING_STATE) {
        this.PromiseState = PROMISE_REJECTED_STATE;
        this.PromiseResult = reason;
        
        setTimeout(() => {
          this.callbacks.forEach(cb => {
            cb.onRejected(this.PromiseResult);
          })
        })
      }
    }
    then(onResolved, onRejected) {
      if (typeof onResolved !== 'function') {
        onResolved = res => res
      }
      if (typeof onRejected !== 'function') {
        onRejected = reason => throw reason
      }
      return new MyPromise((resolve, reject) => {
        const cb = fn => {
          try {
            const res = fn(this.PromiseResult);
            if (res instanceof MyPromise) {
              res.then(
                res => { resolve(res) }
              ),
              err => { reject(err) }
            }
            
          } catch(error) {
            reject(error)
          }
        };
        if (this.PromiseState === PROMISE_FULFILLED_STATE) {
          setTimeout(() => {
            cb(onResolved);
          })
        }
        if (this.PromiseState === PROMISE_REJECTED_STATE) {
          setTimeout(() => {
            cb(onRejected);
          })
        }
        if (this.PromiseState === PROMISE_PENDING_STATE) {
          this.callbacks.push({
            onResolved: () => {
              cb(onResolved);
            },
            onRejected: () => {
              cb(onRejected);
            }
          })
        }
      });
    }
  }
  catch(onRejected) {
    return this.then(null, onRejected);
  }
  
  static resolve(result) {
    return new MyPromise((resolve, reject) => {
      if (result instanceof MyPromise) {
        result.then(
          (res) => { resolve(res) },
          (err) => { reject(err) } 
        )
      } else {
          resolve(result);
      }
    });
  }
  
  static reject(reason) {
    return new MyPromise((resolve, reject) => {
      reject(reason);
    })
  }
  
  static all(pros) {
    let result = [];
    return new MyPromise((resolve, reject) => {
      pros.forEach((item, index) => {
        item.then(
          res => {
            result[index] = res;
            if (result.length === pros.length) {
              resolve(result)
            }
          },
          err => {
            reject(err);
          }
        )
      })
    })
  }
  
  static race(pros) {
    return new MyPromise((resolve, reject) => {
      pros.forEach((item, index) => {
        item.then(
          (res) => { resolve(res) },
          (err) => { reject(err) }
        )
      })
    })
  }
}
```

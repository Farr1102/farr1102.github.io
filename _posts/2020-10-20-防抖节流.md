### 防抖

```js
function debounce(fn, delay) {
  let timer;
  return function() {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, arguments);
    }, delay);
  }
}
```

### 节流

## 通过定时器

```js
function throttle(fn, delay) {
  let timer;
  return function() {
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(this, arguments);
        timer = null;
      }, delay);
    }
  }
}
```

## 通过时间戳

```js
function throttle(fn, delay) {
  let begin = new Date().getTime();
  return function() {
    let curr = new Date().getTime();
    if (curr - begin > delay) {
      fn.apply(this, arguments);
    }
    begin = curr
  }
}
```
